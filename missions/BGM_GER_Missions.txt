	generic = no
	ai = yes

		OR = {
			tag = GER
			tag = HLR
		}
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	##Mission Start
	GER_economy = {
		icon = mission_early_game_buildings
		required_missions = {
			
		}
		position = 1
		provinces_to_highlight = {
			##PROVINCESHIGHLIGHTCODE
		}
		trigger = {
			stability = 0
			OR = {
				full_idea_group = infrastructure_ideas
				full_idea_group = economic_ideas
				full_idea_group = trade_ideas
			}
		}
		effect = {
			custom_tooltip = GER_eco_policy_tooltip

			add_adm_power = 100

			add_country_modifier = {
				name = ger_economic_policy
				duration = 10950
			}

			if = {
				limit = {
					government = republic
					OR = {
						bgm_rev = yes
						bgm_comgovt = yes
						has_reform = peasants_republic
					}
				}
				set_country_flag = welfare_state
			}
			else_if = {
				limit = { 
					government = republic
					NOT = {
						OR = {
							bgm_rev = yes
							bgm_comgovt = yes
							has_reform = peasants_republic
						}
					}
				}
				set_country_flag = corporatist_state
			}
			else = {
				set_country_flag = militarist_state
			}

			swap_non_generic_missions = yes
		}
	}
}
