lollard_ref_trigger = { # Lollardy
	OR = {
		calc_true_if = {
			amount = 10
			all_country = {
				OR = {
					religion = lollard
					religion = waldensian
					religion = hussite
					religion = cathar
					religion = fraticelli
				}
			}
		}
		calc_true_if = {
			any_province = {
				religion = lollard
				is_reformation_center = yes
			}
			amount = 3
		}
	}
	#num_of_cities = 5
	religion = lollard
	#has_friendly_reformation_center = yes
	NOT = { has_global_flag = preprot_heresy }
}
