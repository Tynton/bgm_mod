emancipate_slaves = {
	category = 4

	allow = {
		has_country_flag = will_emancipate
		NOT = { has_country_flag = freed_slaves }
	}
	
	effect = {
		set_country_flag = freed_slaves
		random_list = {
			75 = {
				every_owned_province = {
					limit = {
						OR = {
							region = southeast_america_region
							region = rio_grande_region
							area = west_florida_area
							area = lower_louisiana_area
							area = upper_louisiana_area
							area = ozarks_area
							area = choctaw_area
						}
					}
					set_province_flag = csa
					cede_province = CSA
					add_core = CSA
				}
				CSA = {
					add_years_of_income = 1
					add_yearly_manpower = 1
					add_army_tradition = 50
					create_general = { tradition = 100 }
					country_event = {
						id = americangovt.8
						days = 270
					}
				}
			}
			25 = {
				every_owned_province = {
					limit = {
						OR = {
							region = southeast_america_region
							region = mississippi_region
							region = rio_grande_region
						}
						OR = {
							trade_goods = tobacco
							trade_goods = cotton
						}
					}
					spawn_rebels = {
						size = 3
						type = particularist_rebels
					}
				}
			}
		}
		
	}
	
	modifier = {
		global_autonomy = -0.05
		republican_tradition = 1
	}
	backing_percentage = 25
	
	chance = {
		factor = 1
	}

	}
}
