#Province Cultural Assimilation

country_event = {
	id = national_program.5
	title = "national_program.n.5"
	desc = "national_program.d.5"
	picture = MODERN_UNREST_eventPicture
	
	#hidden = yes
	
	trigger = {
		has_country_modifier = basic_language_ed
		
		#any_neighbor_province = {
		#	culture_group = ROOT
		#}
		
		any_owned_province = {
			NOT = {
				culture_group = ROOT
			}
		}
		OR = {
			stability = 2
			any_owned_province = {
				is_prosperous = yes
			}
		}
	}
	
	mean_time_to_happen = {
		months = 124
		
		modifier = {
			factor = 0.25
			all_owned_province = {
				is_prosperous = yes
			}
		}

		modifier = {
			factor = 0.75
			stability = 1
		}

		modifier = {
			factor = 0.75
			stability = 3
		}
		
		#modifier = {
		#	factor = 0.5
		#	any_neighbor_province = {
		#		culture_group = ROOT
		#	}
		#}
	}
	
	option = {
		name = "national_program.5.a"
		random_owned_province = {
			limit = {
				OR = {
					any_neighbor_province = {
						culture_group = ROOT
					}
					has_port = yes
				}
				is_prosperous = yes
				NOT = {
					culture_group = ROOT
				}
			}
			change_culture = ROOT
		}
	}
}

#Province changes culture

country_event = {
	id = national_program.10
	title = none
	desc = none
	picture = BUREAU_eventPicture

	hidden = yes
	
	trigger = {
		OR = {
			has_country_modifier = standard_language_adm
			has_reform = intolerance_reform
		}
		any_owned_province = {
			OR = {
				has_province_modifier = standardized_language_french
				has_province_modifier = standardized_language_italian
				has_province_modifier = standardized_language_germanic
				has_province_modifier = standardized_language_spanish
			}
			has_owner_accepted_culture = yes
			any_neighbor_province = {
				culture_group = ROOT
			}
		}
	}
	
	mean_time_to_happen = {
		months = 39

		modifier = {
			factor = 0.6
			stability = 1
		}

		modifier = {
			factor = 0.5
			stability = 2
		}

		modifier = {
			factor = 0.25
			has_reform = intolerance_reform
		}

		modifier = {
			factor = 2.0
			NOT = { has_reform = intolerance_reform }
		}

		modifier = {
			factor = 0.25
			culture_group = ROOT
		}
	}
	
	option = {
		name = none
		random_owned_province = {
			limit = {
				NOT = {
					culture_group = ROOT
				}
				OR = {
					has_province_modifier = standardized_language_french
					has_province_modifier = standardized_language_italian
					has_province_modifier = standardized_language_germanic
					has_province_modifier = standardized_language_spanish
				}
				has_owner_accepted_culture = yes
				any_neighbor_province = {
					culture_group = ROOT
				}
			}
			change_culture = ROOT
		}
	}
}
