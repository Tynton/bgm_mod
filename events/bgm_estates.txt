	}
}

				}

					define_leader_to_ruler = {
						type = general

# jingoism
province_event = {
	id = bgm_estates.8
	title = "bgm_estates.8.t"
	desc = "bgm_estates.8.d"
	picture = MIL_ADVANCE_eventPicture

	trigger = {
		any_neighbor_province = {
			owner = {
				NOT = {
					OR = {
						is_subject_of = ROOT
						alliance_with = ROOT
					}
				}
			}
			OR = {
				is_claim = owner
				is_core = owner
			}
			NOT = {
				owned_by = ROOT
			}
		}
		owner = {
			has_estate = estate_military
		}
	}

	mean_time_to_happen = {
		months = 250

		modifier = {
			factor = 0.75
			owner = {
				religion_group = christian
			}
			any_neighbor_province = {
				owner = {
					NOT = {
						religion_group = christian
					}
				}
			}
		}
	}
	
	option = {
		name = "bgm_estates.8.a" #start border skirmishes
		owner = {
			add_yearly_manpower = -0.05
			add_estate_loyalty = {
				estate = estate_military
				loyalty = 10
			}
		}
		random_neighbor_province = {
			limit = {
				NOT = {
					OR = {
						owned_by = ROOT
						owner = {
							OR = {
								is_subject_of = ROOT
								alliance_with = ROOT
							}
						}
					}
				}
			}
			owner = {
				country_event = {
					id = bgm_estates.9
				}
			}
		}
	}

	option = {
		name = "bgm_estates.8.b" #don't
		owner = {
			add_estate_loyalty = {
				estate = estate_military
				loyalty = -10
			}	
		}
	}
}

# jingoism pt 2
country_event = {
	id = bgm_estates.9
	title = "bgm_estates.9.t"
	desc = "bgm_estates.9.d"
	picture = MIL_ADVANCE_eventPicture

	is_triggered_only = yes

	immediate = {
		add_casus_belli = {
			type = cb_border_war
			target = FROM
		}
		add_opinion = {
			who = FROM
			modifier = border_aggression
		}
	}
	
	option = {
		name = "bgm_estates.9.a" #respond in kind
		add_yearly_manpower = -0.075
		FROM = {
			add_casus_belli = {
				type = cb_border_war
				target = ROOT
			}
			add_opinion = {
				who = ROOT
				modifier = border_aggression
			}
		}

		ai_chance = {
			factor = 5
		}
	}

	option = {
		name = "bgm_estates.9.b" #concede the territory
		random_owned_province = {
			limit = {
				owned_by = ROOT
				any_neighbor_province = {
					owned_by = FROM
				}
			}
			cede_province = FROM
		}

		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				OR = {
					bgm_comgovt = yes
					has_reform = fascist_mechanic
					has_reform = futurist_mechanic
					has_reform = authoritarian_mechanic
				}
			}
		}
	}
}
