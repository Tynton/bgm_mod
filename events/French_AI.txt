country_event = {
	id = French_AI.1
	title = "French_AI.n.1"
	desc = "French_AI.d.1"
	picture = BUREAU_eventPicture
	
	hidden = yes
	
	trigger = {
		is_subject = no
		NOT = {
			has_opinion_modifier = {
				who = FRA
				modifier = aggressive_expansion
				value = 35
			}
		}
		OR = {
			AND = {
				any_owned_province = {
					is_claim = FRA
				}
				ai = yes
			}
			FRA = {
				has_casus_belli = {
					type = cb_imperial
					target = ROOT
				}
			}
		}
		FRA = {
			is_subject = no
			manpower_percentage = 0.9
			army_size_percentage = 0.9
			NOT = { num_of_loans = 1 }
			ai = yes
			army_strength = {
				who = ROOT
				value = 2.25
			}
			is_at_war = no
			NOT = { truce_with = ROOT }
		}
		OR = {
			AND = {
				any_owned_province = {
					is_claim = FRA
				}
				capital_scope = {
					continent = europe
				}
			}
			AND = {
				is_neighbor_of = FRA
				capital_scope = {
					continent = europe
				}
				FRA = {
					has_country_flag = fra_europe
				}
			}
		}
		NOT = {
			any_ally = {
				army_strength = {
					who = FRA
					value = 0.5
				}
			}
		}
		NOT = {
			AND = {
				army_strength = {
					who = FRA
					value = 0.5
				}
				truce_with = FRA
			}
		}
		NOT = {
			any_country = {
				has_guaranteed = ROOT
			}
		}
		NOT = {
			alliance_with = FRA 
		}
		OR = {
			is_part_of_hre = no
			AND = {
				is_part_of_hre = yes
				emperor = {
					NOT = {
						army_strength = {
							who = FRA
							value = 0.35
						}
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		years = 2

		modifier = {
			factor = 0.33
			capital_scope = {
				region = italy_region
			}
		}

		modifier = {
			factor = 0.5
			NOT = {
				military_strength = {
					who = FRA
					value = 0.25
				}
			}
		}
		modifier = {
			factor = 0.1
			NOT = {
				military_strength = {
					who = FRA
					value = 0.1
				}
			}
		}
		modifier = {
			factor = 0.1
			NOT = {
				any_country = {
					alliance_with = ROOT
				}
			}
		}
	}
	
	#he decs
	
	option = {
		name = none
		
		FRA = {
			if = {
				limit = {
					has_casus_belli = {
						type = cb_imperial
						target = ROOT
					}
				}
				declare_war_with_cb = {
					casus_belli = cb_imperial
					who = ROOT
				}
			}
			else = {
				declare_war_with_cb = {
					who = ROOT
					casus_belli = cb_bgmai_conquest
				}
			}
		}
	}
}
