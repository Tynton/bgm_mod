		}
	}

#attacks for union

country_event = {
	id = Austria_AI.2
	title = none
	desc = none
	picture = none
	
	hidden = yes
	
	trigger = {
		HAB = {
			ai = yes
			OR = {
				num_of_allies = 2
				alliance_with = HUN
			}
			has_casus_belli = {
				type = cb_restore_personal_union
				target = ROOT
			}
			is_at_war = no
		}
		ROOT = {
			NOT = {
				any_ally = {
					army_strength = {
						who = HAB
						value = 0.5
					}
				}
				
				army_strength = {
					who = HAB
					value = 0.5
				}
			}
			NOT = {
				any_country = {
					has_guaranteed = ROOT
				}
			}
		}
		NOT = {
			HAB = {
				truce_with = ROOT
			}
		}
	}

	mean_time_to_happen = {
		months = 48
		modifier = {
			factor = 0.01
			NOT = {
				any_country = {
					alliance_with = ROOT
				}
			}
		}
	}
	
	#austria attacks
	
	option = {
		name = "Austria_AI.2.a"
		
		HAB = {
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_restore_personal_union
			}
		}
	}
}

#ally electors - no longer specific to austria but is now for all emperors

country_event = {
	id = Austria_AI.3
	title = none
	desc = none
	picture = none
	
	hidden = yes
	
	trigger = {
		emperor = {
			num_of_free_diplomatic_relations = 1
			ai = yes
			is_emperor = yes
			NOT = { truce_with = ROOT }
			NOT = {
				OR = {
					is_rival = ROOT
					is_enemy = ROOT
				}
			}
			NOT = {
				calc_true_if = {
					all_country = {
						is_elector = yes
						alliance_with = emperor
					}
					amount = 2
				}
			}
		}
		ROOT = {
			num_of_free_diplomatic_relations = 1
			ai = yes
			is_elector = yes 
			NOT = { truce_with = emperor }
			NOT = {
				has_country_flag = allied_emperor
			}
			NOT = {
				OR = {
					is_rival = emperor
					is_enemy = emperor
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 5
	}
	
	#abdicates
	
	option = {
		name = "Austria_AI.3.a"
		
		emperor = {
			create_alliance = ROOT
		}
		
		ROOT = {
			set_country_flag = allied_emperor
		}
	}
}
