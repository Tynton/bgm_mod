	title = "BGM_BUR_missions.3.t"
	desc = "BGM_BUR_missions.3.d"
	picture = GOOD_WITH_MONARCH_eventPicture

	major = yes

	is_triggered_only = yes

	immediate = {
		kill_ruler = yes
	}
	
	option = {
		name = "BGM_BUR_missions.3.a" #Elect the Burgundian Candidate
		BUR = {
			create_subject = {
				subject_type = vassal
				subject = LIE
			}
		}
		define_ruler = {
			culture = burgundian
		}
		add_liberty_desire = -50

		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				NOT = {
					has_opinion = {
						who = BUR
						value = -25
					}
				}
			}
		}
	}
	
	option = {
		name = "BGM_BUR_missions.3.b" #Resist their advances
		BUR = {
			add_casus_belli = {
				type = cb_vassalize_mission
				target = LIE
			}
		}

		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				has_opinion = {
					who = BUR
					value = -25
				}
			}
		}
	}
}
