	title = "world_war_events.n.3"

	trigger = {
		current_age = age_of_great_wars 
		
		is_great_power = yes 
		
		is_at_war = no
		
		NOT = {
			any_rival_country = {
				OR = {
					war_with = ROOT
					truce_with = ROOT
				}
			}
			has_country_flag = diplomatic_crisis
		}
		any_rival_country = {
			is_neighbor_of = ROOT
			OR = {
				num_of_allies = 2
				num_of_subjects = 4
			}
			is_great_power = yes
		}
		
		OR = {
			num_of_allies = 2
			num_of_subjects = 4
		}
		
		NOT = {
			has_global_flag = world_war_ongoing
		}
		
		NOT = {
			OR = {
				ROOT = {
					OR = {
						technology_group = modern_nuke
						technology_group = modern_german_nuke
						technology_group = modern_soviet_nuke
					}
				}
				FROM = {
					OR = {
						technology_group = modern_nuke
						technology_group = modern_german_nuke
						technology_group = modern_soviet_nuke
					}
				}
			}
		}

		#all_country = {
		#	limit = {
		#		is_great_power = yes
		#	}
		#	is_at_war = no
		#}
	}

	mean_time_to_happen = {
		months = 220
		
		modifier = {
			factor = 0.9
			is_year = 1910
		}
		
		modifier = {
			factor = 0.75
			has_reform = fascist_mechanic
		}
	}
	
	#escalate tension
	
	option = {
		name = "world_war_events.3.a"
		random_rival_country = { 
			country_event = {
				id = world_war_events.4
				days = 30
			}
		} 
		
		ai_chance = { factor = 15 }
	}
	
	#settle peacefully
	
	option = {
		name = "world_war_events.3.b"

		add_stability = -1

	}
}
