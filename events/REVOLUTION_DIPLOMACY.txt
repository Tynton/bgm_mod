country_event = {
	id = REVOLUTION_DIPLOMACY.3
	title = "REVOLUTION_DIPLOMACY.n.3"
	desc = "REVOLUTION_DIPLOMACY.d.3"
	picture = MILITARY_CAMP_eventPicture
	
	major = yes
	
	trigger = {
		current_age = age_of_revolutions 

		OR = {
			is_great_power = yes
			any_country = {
				is_great_power = yes
				any_country = {
					war_with = PREV
					is_revolution_target = yes
				}
			}
		}

		same_continent = ROOT
		
		is_at_war = no
		
		OR = {
			stability = 1
			has_any_disaster = no
		}
		
		NOT = {
			any_country = {
				OR = {
					war_with = ROOT
					truce_with = ROOT
				}
				NOT = {
					great_power_rank = 3
				}
				is_revolution_target = yes
				
			}
			has_global_flag = diplomatic_crisis
		}
		is_great_power = yes
		
		bgm_rev = no
		
		OR = {
			any_country = {

				is_neighbor_of = ROOT
				
				is_revolution_target = yes
				NOT = {
					truce_with = ROOT
				}
				is_hegemon = yes
				NOT = { 
					capital_scope = {
						region = british_isles_region
					} 
				}
			}
			
			any_rival_country = {

				is_revolution_target = yes
				NOT = {
					truce_with = ROOT
				}
				is_hegemon = yes
			}
		}
	}

	mean_time_to_happen = {
		months = 48

		modifier = {
			factor = 0.75
			has_reform = fascist_mechanic
		}
	}
	
	#escalate tension
	
	option = {
		name = "REVOLUTION_DIPLOMACY.3.a"
		random_country = { 
			limit = {
				is_revolution_target = yes
			}
			country_event = {
				id = REVOLUTION_DIPLOMACY.4
				days = 15
			}
		} 
		set_global_flag = diplomatic_crisis
		
		ai_chance = { factor = 50 }
	}
	
	#settle peacefully
	
	option = {
		name = "REVOLUTION_DIPLOMACY.3.b"
		add_prestige = -10
		add_stability = -1
		
		ai_chance = { factor = 1 }
	}
}

country_event = {
	id = REVOLUTION_DIPLOMACY.6
	title = "REVOLUTION_DIPLOMACY.n.6"
	desc = "REVOLUTION_DIPLOMACY.d.6"
	picture = MILITARY_CAMP_eventPicture
	trigger = {
		is_revolution_target = yes
		current_age = age_of_revolutions
		NOT = {
			has_country_flag = rev_dip
		}
	}

	mean_time_to_happen = {
		days = 1

		modifier = {
			factor = 0.001
			is_revolution_target = yes
		}
	}

	option = {
		name = "REVOLUTION_DIPLOMACY.6.a"
		
		
		hidden_effect = {
			every_country = {
				limit = {
					bgm_rev = no
				}
				add_opinion = {
					who = ROOT
					modifier = embraced_revolution
					years = 15
				}
			}
			
			set_country_flag = rev_dip

			every_country = {
				limit = {
					bgm_rev = no
				}
				add_trust = {
					who = ROOT
					value = -50
					mutual = yes
				}
			}
		}
	}
}
