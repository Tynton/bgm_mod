#initial italian revolt

country_event = {
	id = italyunity.1
	title = "italyunity.n.1"
	desc = "italyunity.d.1"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes

	trigger = {
		is_year = 1815
		AND = {
			development_in_provinces = {
				value = 500
				region = italy_region
			}
			capital_scope = {
				continent = europe
			}
			NOT = {
				exists = ITA
				tag = ITA
				stability = 1
				culture_group = latin
			}
		}
	}	

	mean_time_to_happen = {
		months = 60
	}
	
	option = {
		name = "italyunity.1.a"
		release = ITA
		ITA = {
			change_government = monarchy
			add_government_reform = autocracy_reform
			add_government_reform = parliamentary_reform
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
		}
	}
	
	option = {
		name = "italyunity.1.b"
		release = ITA
		ITA = {
			change_government = monarchy
			add_government_reform = autocracy_reform
			add_government_reform = parliamentary_reform
			create_alliance = ROOT
		}
	}
}

		NOT = { tag = ITF }

		}

		}

			region = italy_region
		}
		
	}

	}

country_event = {
	id = italyunity.6
	title = "italyunity.n.6"
	desc = "italyunity.d.6"
	picture = HRE_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = "italyunity.6.a"
		set_country_flag = preeminent_sisterrepublic
		if = {
			limit = {
				NOT = {
					primary_culture = venetian
					has_country_flag = monaco_c
					OR = {
						exists = ITA
						exists = VIY
						exists = MIT
					}
				}
			}
			change_tag = ITA
		}
		if = {
			limit = {
				NOT = { primary_culture = venetian }
				has_country_flag = monaco_c
				NOT = {
					OR = {
						exists = ITA
						exists = VIY
						exists = MIT
					}
				}
			}
			change_tag = VIY
		}
		if = {
			limit = {
				primary_culture = venetian
				NOT = {
					OR = {
						exists = ITA
						exists = VIY
						exists = MIT
					}
				}
			}
			change_tag = VIY
		}

		if = {
			limit = {
				exists = ITA
				OR = {
					AND = {
						exists = VIY
						VIY = {
							is_subject_of = ROOT
						}
					}
					AND = {
						exists = MIT
						MIT = {
							is_subject_of = ROOT
						}
					}
				}
			}
			ITA = {
				inherit = VIY
				inherit = MIT
			}
		}
	}
}
