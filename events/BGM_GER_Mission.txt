			}

				add_base_production = 1

		}

		if = {
			limit = { 
				any_owned_province = {
					is_capital = no
					province_has_center_of_trade_of_level = 1
				}
			}
			random_owned_province = {
				limit = { 
					is_capital = no
					province_has_center_of_trade_of_level = 1
				}

					name = ger_eco_coop
					duration = -1
				}
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
			}
		}
	}
