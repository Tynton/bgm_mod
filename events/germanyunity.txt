#nations join germany

country_event = {
	id = germanyunity.4
	title = "germanyunity.n.4"
	desc = "germanyunity.d.4"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		exists = GER
		OR = {
			AND = {
				OR = {
					culture_group = germanic
					AND = {
						culture_group = benelux
						is_revolutionary = yes
					}
				}
				NOT = {
					total_development = 500
				}
			}
		}
		NOT = { tag = GER }
		NOT = { tag = GRD }
		NOT = { tag = GRF }
		NOT = { tag = GRC }
		NOT = {
			has_country_flag = denied_annexation
		}
		OR = {
			is_subject = no
			is_subject_of = GER
		}
		GER = { 
			is_subject = no 
			has_institution = nationalism_institution
		}
		NOT = {
			truce_with = GER
		}
		capital_scope = {
			OR = {
				region = north_german_region
				region = south_german_region
			}
		}
	}	

	mean_time_to_happen = {
		months = 3
	}
	
	option = {
		name = "germanyunity.4.a"
		GER = {
			inherit = ROOT
			ROOT = {
				every_owned_province = {
					add_core = GER
				}
			}
		}
		ai_chance = { 
			factor = 1 
			
			modifier = {
				factor = 500
				is_revolutionary = yes
			}
		}
	}
	
	option = {
		name = "germanyunity.4.b"
		GER = {
			ROOT = {
				every_owned_province = {
					add_core = GER
				}
			}
		}
		
		set_country_flag = denied_annexation
		
		ai_chance = { 
			factor = 1 
			
			modifier = {
				factor = 10000
				OR = {
					culture = austrian
					culture = swiss
				}
				is_revolutionary = no
			}
		}
	}
}

country_event = {
	id = germanyunity.9
	title = "germanyunity.9.t"
	desc = "germanyunity.9.d"
	picture = HRE_eventPicture
	
	fire_only_once = yes
	
	major = yes

	is_triggered_only = yes
	
	option = {
		name = "germanyunity.9.a"
		hidden_effect = {
			every_country = {
				limit = {
					OR = {
						is_subject_of_type = german_confederation_member
						has_country_flag = gercon_member
					}
					OR = {
						culture_group = benelux
						
						culture_group = germanic
						
						primary_culture = czech
						primary_culture = moravian
						primary_culture = schlesian
					}
				}
				random_country = {
					limit = {
						has_country_flag = victorious_zollverein
					}
					create_subject = {
						subject_type = zollverein
						subject = PREV
					}
					PREV = {
						add_country_modifier = {
							name = zollverein_subject
							duration = -1
						}
					}
				}
			}
		}
		custom_tooltip = zollverein_tooltip
	}
}

#Zollverein
country_event = {
	id = germanyunity.10
	title = "germanyunity.10.t"
	desc = "germanyunity.10.d"
	picture = HRE_eventPicture
	is_triggered_only = yes
	
	option = {
		name = "germanyunity.10.a"
		random_country = {
			limit = {
				has_country_flag = victorious_zollverein
			}
			create_subject = {
				subject_type = zollverein
				subject = ROOT
			}
		}

		ROOT = {
			add_country_modifier = {
				name = zollverein_subject
				duration = -1
			}
		}

		if = {
			limit = {
				tag = FROM
				NOT = {
					any_subject_country = {
						is_subject_of_type = zollverein
					}
				}
			}
			add_country_modifier = {
				name = zollverein_overlord
				duration = -1
			}
		}


		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				FROM = {
					NOT = {
						OR = {
							ai_attitude = {
								who = ROOT
								attitude = attitude_threatened
							}
							ai_attitude = {
								who = ROOT
								attitude = attitude_allied
							}
							ai_attitude = {
								who = ROOT
								attitude = attitude_friendly
							}
						}
					}
				}
			}
		}
	}

	option = {
		name = "germanyunity.10.b"
		FROM = {
			add_opinion = {
				who = ROOT
				modifier = no_zollverein
			}
		}


		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				FROM = {
					OR = {
						ai_attitude = {
							who = ROOT
							attitude = attitude_threatened
						}
						ai_attitude = {
							who = ROOT
							attitude = attitude_allied
						}
						ai_attitude = {
							who = ROOT
							attitude = attitude_friendly
						}
					}
				}
			}
		}
	}
}
