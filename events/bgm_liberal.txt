namespace = bgm_liberal

# Coup
country_event = {
	id = bgm_liberal.1
	title = "bgm_liberal.1.t"
	desc = "bgm_liberal.1.d"
	picture = MARCH_eventPicture

	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { 
				has_reform = futurist_mechanic
				bgm_comgovt = yes
			}
			set_country_flag = communism_lib
		}
		if = {
			limit = { 
				has_reform = futurist_mechanic
				bgm_comgovt = no
			}
			set_country_flag = futurism_lib
		}
		if = {
			limit = { 
				has_reform = authoritarian_mechanic
			}
			set_country_flag = auth_lib
		}
		if = {
			limit = { 
				has_reform = fascist_mechanic
			}
			set_country_flag = fasc_lib
		}
	}
	
	option = {
		name = "bgm_liberal.1.a" #fight
		add_stability = -4
		capital_scope = {
			spawn_rebels = {
				size = 3
				type = democratic_rebels
			}
		}
		if = {
			limit = { 
				num_of_cities = 10
			}
			random_owned_province = {
				limit = { 
					NOT = { 
						is_capital_of = ROOT 
					} 
				}
				spawn_rebels = {
					size = 2
					type = democratic_rebels
				}
			}
		}
		if = {
			limit = { 
				num_of_cities = 15
			}
			random_owned_province = {
				limit = { 
					NOT = { 
						is_capital_of = ROOT 
					} 
				}
				spawn_rebels = {
					size = 1
					type = democratic_rebels
				}
			}
		}
		if = {
			limit = { 
				num_of_cities = 20
			}
			random_owned_province = {
				limit = { 
					NOT = { 
						is_capital_of = ROOT 
					} 
				}
				spawn_rebels = {
					size = 2
					type = democratic_rebels
				}
			}
			random_owned_province = {
				limit = { 
					NOT = { 
						is_capital_of = ROOT 
					} 
				}
				spawn_rebels = {
					size = 1
					type = democratic_rebels
				}
			}
		}

		ai_chance = {
			factor = 1

			modifier = {
				factor = 25
				adm_power = 400
				army_size_percentage = 0.5
			}
		}
	}
	
	option = {
		name = "bgm_liberal.1.b" #Concede
		kill_ruler = yes
		change_government = democracy_govt
		add_government_reform = parliamentary_republic_reform_bgm

		add_adm_power = -250
		add_dip_power = -250
		add_mil_power = -250
		add_prestige = -50

		add_stability = 1

		#end_disaster = bgm_liberal

		if = {
			limit = {
				NOT = {
					is_subject_of_type = pact_member
				}
			}
			overlord = {
				add_casus_belli = {
					target = ROOT
					type = cb_annex
					months = 120
				}
			}
			grant_independence = yes
			end_disaster = bgm_liberal
		}





		ai_chance = {
			factor = 1

			modifier = {
				factor = 1000
				OR = {
					liberty_desire = 50
					NOT = {
						has_opinion = {
							who = overlord
							value = -25
						}
					}
				}
			}
		}
	}

	after = {
		add_yearly_manpower = -0.25
	}
}

country_event = {
	id = bgm_liberal.4
	title = "bgm_liberal.4.t"
	desc = "bgm_liberal.4.d"
	picture = BUREAU_eventPicture

	trigger = {
		has_disaster = bgm_liberal
		has_reform = democratic_mechanic
	}

	mean_time_to_happen = {
		months = 6
	}

	immediate = {
		set_country_flag = had_countercoup
	}
	
	
	option = {
		name = "bgm_liberal.4.a" #fight
		trigger = {
			stability = 1
		}

		if = {
			limit = {
				has_country_flag = communism_lib
			}
			capital_scope = {
				spawn_rebels = {
					size = 3
					type = socialist_rebels
				}
			}
		}
		if = {
			limit = {
				has_country_flag = futurism_lib
			}
			capital_scope = {
				spawn_rebels = {
					size = 3
					type = futurist_rebels
				}
			}
		}
		if = {
			limit = {
				has_country_flag = auth_lib
			}
			capital_scope = {
				spawn_rebels = {
					size = 1
					type = fascist_rebels
				}
			}
		}
		if = {
			limit = {
				has_country_flag = fasc_lib
			}
			capital_scope = {
				spawn_rebels = {
					size = 3
					type = fascist_rebels
				}
			}
		}
		
		ai_chance = {
			factor = 1

			modifier = {
				factor = 0
				NOT = {
					army_size_percentage = 0.5
				}
			}
		}
	}
	
	option = {
		name = "bgm_liberal.4.b" #the coup succeeds

		add_stability = -2

		if = {
			limit = {
				has_country_flag = communism_lib
			}
			change_government = futurist_government
			add_government_reform = socialist_republic
		}
		if = {
			limit = {
				has_country_flag = futurism_lib
			}
			change_government = futurist_government
			add_government_reform = futurist_government_reform
		}
		if = {
			limit = {
				has_country_flag = auth_lib
			}
			change_government = authoritarian_government
			add_government_reform = authoritarian_mechanic
		}
		if = {
			limit = {
				has_country_flag = fasc_lib
			}
			change_government = fascist_government
			add_government_reform = fascist_government_reform
		}

		ai_chance = {
			factor = 1
			
			modifier = {
				factor = 0
				army_size_percentage = 0.5
			}
		}
	}
}
