#DISSOLUTION OF THE REVOLUTION
province_event = {
	id = alt_center_rev.2
	title = "alt_center_rev.n.2"
	desc = "alt_center_rev.d.2"
	picture = ANGRY_MOB_eventPicture
	
	#fire_only_once = yes
	
	hidden = yes

	#major = yes

	trigger = {
		NOT = { current_age = age_of_revolutions } #is_institution_enabled = nationalism_institution
		any_province = {
			is_center_of_revolution = yes
		}
		#is_center_of_revolution = yes
	}	

	mean_time_to_happen = {
		days = 2
	}
	
	option = {
		name = "alt_center_rev.2.a"
		owner = {
			remove_center_of_revolution = yes
		}
		hidden_effect = {
			every_province = {
				limit = { has_revolution_in_province = yes }
				remove_revolution_from_province_effect = yes
			}
		}
		every_country = {
			limit = {
				is_revolution_target = yes
			}
			country_event = {
				id = alt_center_rev.4
			}
		}
	}
}
