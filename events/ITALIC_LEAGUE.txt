country_event = {
	id = italic_league.3
	title = "italic_league.n.3"
	desc = "italic_league.d.3"
	picture = HRE_eventPicture
	
	major = yes
	
	fire_only_once = yes

	trigger = {
		any_country = {
			OR = {
				AND = {
					NOT = { culture_group = latin }
					calc_true_if = {
						all_owned_province = {
							OR = {
								culture_group = latin
								region = italy_region
							}
						}
						amount = 5
					}
				}
				capital_scope = {
					OR = {
						region = france_region
						region = balkan_region
					}
				}
			}
			total_development = 300
		}
		is_subject = no
		NOT = { has_global_flag = italea_happened }
		has_global_flag = italians_leave_hre
		
		NOT = {
			OR = {
				any_country = {
					is_subject_of_type = italic_league_member
				}
				any_country = {
					is_subject_of_type = italic_league_member
				}
				has_country_flag = no_italea
			}
		}
		OR = {
			culture_group = latin
			tag = SAV
		}
		OR = {
			total_development = 66
			capital_scope = {
				is_institution_origin = renaissance
			}
		}
		NOT = { OR = { exists = ITA exists = VIY exists = MIT } }
	}

	mean_time_to_happen = {
		months = 8
		
		modifier = {
			factor = 0.7
			OR = {
				france_region = {
					any_country = { 
						total_development = 450
					}
				}
				iberia_region = {
					any_country = {
						total_development = 450
					}
				}
			}
		}
	}
	
	
	option = {
		name = "italic_league.3.a"

		set_global_flag = italea_happened
		
		set_country_flag = italea_leader

		hidden_effect = {
			if = {
				limit = { 
					ai = yes
				}
				add_country_modifier = {
					name = italea_ai
					duration = -1
				}
				remove_advisor_by_category = adm
				remove_advisor_by_category = dip
				remove_advisor_by_category = mil
			}
		}
		
		every_country = {
			limit = {
				OR = {
					culture_group = latin
					culture_group = franco_provencal
				}
				capital_scope = {
					OR = {
						region = italy_region
						region = france_region
					}
				}
				#is_at_war = no
				is_subject = no
			}
			#total_development = 225
			country_event = {
				id = italic_league.4
			}
		}
		#country_event = { #40 years to reform to integrate
		#	id = italic_league.5
		#	days = 14600
		#}
	}

	option = {
		name = "italic_league.3.b"
		
		set_country_flag = no_italea
		
		add_prestige = -50
		add_stability = 2

		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				stability = 1
			}
		}
	}
}

country_event = {
	id = italic_league.4
	title = "italic_league.n.4"
	desc = "italic_league.d.4"
	picture = HRE_eventPicture
	
	is_triggered_only = yes

	option = { #Join
		name = "italic_league.4.a"
		
		trigger = {
			is_subject = no
			
			NOT = {
				OR = {
					dynasty = "de Trastámara"
					has_country_flag = italea_leader
				}
			}
		}
		
		#release_all_subjects = yes
		
		FROM = {
			if = {
				limit = {
					NOT = {
						has_subject_of_type = italic_league_member
					}
				}
				add_country_modifier = {
					name = italea_1
					duration = -1
				}
			}
			create_subject = {
				subject_type = italic_league_member
				subject = ROOT
			}
			
		}
		
		ai_chance = { 
			factor = 1
			modifier = {
				factor = 0
				OR = {
					NOT = {
						has_opinion = {
							who = FROM
							value = 0
						}
					}
					military_strength = {
						who = FROM
						value = 1
					}
				}
			}
		}
	}
	
	option = { #Don't join
		name = "italic_league.4.b"
		
		trigger = {
			OR = {
				NOT = {
					capital_scope = {
						region = italy_region
					}
				}
				#ai = no
				is_subject = no
			}
			NOT = {
				has_country_flag = italea_leader
			}
		}
		
		ai_chance = { 
			factor = 1
			modifier = {
				factor = 0
				OR = {
					has_opinion = {
						who = FROM
						value = 0
					}
					NOT = {
						military_strength = {
							who = FROM
							value = 1
						}
					}
				}
			}
		}
	}
	
	option = { #Leader option
		name = "italic_league.4.c"
		
		trigger = {
			has_country_flag = italea_leader
		}
	}
}

country_event = {
	id = italic_league.6
	title = "italic_league.n.6"
	desc = "italic_league.d.6"
	picture = HRE_eventPicture
	
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				has_country_modifier = italea_ai
			}
			remove_country_modifier = italea_ai
		}
		set_country_flag = integrated_italea
	}
	
	option = { #Join
		name = "italic_league.6.a"

		trigger = {
			OR = {
				FROM = {
					OR = {
						has_country_modifier = italea_2
						has_country_modifier = italea_3
						has_country_modifier = italea_4
					}
				}
			}
		}

		if = {
			limit = {
				ai = yes
				NOT = {
					has_country_flag = bigger_than_italea
				}
			}
			FROM = {
				inherit = ROOT
				ROOT = {
					every_owned_province = {
						add_core = FROM
					}
				}
			}
		}
		else = {
			FROM = {
				create_subject = {
					subject_type = vassal
					who = ROOT
				}
			}
		}
		
		ai_chance = { 
			factor = 1 
			
			modifier = {
				factor = 0.001
				total_development = 150
			}
		}
	}
	
	option = { #Leave
		name = "italic_league.6.b"

		trigger = {
			NOT = {
				has_subject_of_type = italic_league_member
			}
			FROM = {
				OR = {
					has_country_modifier = italea_1
					has_country_modifier = italea_2
				}
			}
		}
		FROM = {
			add_casus_belli = {
				target = ROOT
				type = cb_annex
				months = 240
			}
		}
		every_owned_province = {
			add_claim = FROM
		}
		
		grant_independence = yes
		ai_chance = { 
			factor = 0
			
			modifier = {
				factor = 10
				NOT = {
					total_development = 150
				}
			}
		}
	}
	option = { #Leader
		name = "italic_league.6.c"

		trigger = {
			has_country_flag = italea_leader
		}

		if = {
			limit = {
				primary_culture = venetian
			}
			change_tag = VIY
			set_country_flag = venetian_italy
		}

		if = {
			limit = {
				has_country_flag = monaco_c
			}
			change_tag = MIT
			set_country_flag = regular_italy
		}

		if = {
			limit = {
				NOT = { primary_culture = venetian }
			}
			change_tag = ITA
			set_country_flag = regular_italy
		}
		set_country_flag = formed_italy_flag

		if = {
			limit = {
				has_country_modifier = italea_3
			}
			override_country_name = "Italic League"
		}
		

		every_subject_country = {
			limit = {
				is_subject_of_type = italic_league_member
			}
			country_event = {
				id = italic_league.6
			}
		}
		
	}
}
