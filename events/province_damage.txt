namespace = province_damage

province_event = {
	id = province_damage.1
	title = "province_damage.n.1"
	desc = "province_damage.d.1"
	picture = ANGRY_MOB_eventPicture

	hidden = yes
	
	trigger = {
		OR = {
			AND = {
				owner = {
					has_country_modifier = genocides
				}
				has_owner_accepted_culture = no
			}
			AND = {
				controller = {
					mil_tech = 10
				}
				devastation = 75
			}
			AND = {
				controller = {
					mil_tech = 20
				}
				devastation = 50
			}
			AND = {
				controller = {
					mil_tech = 30
				}
				devastation = 25
			}
		}
		base_tax = 2
	}	

	mean_time_to_happen = {
		months = 150
		
		modifier = {
			factor = 0.8
			devastation = 30
		}
		
		modifier = {
			factor = 0.8
			devastation = 40
		}
		
		modifier = {
			factor = 0.8
			devastation = 50
		}

		modifier = {
			factor = 0.8
			devastation = 60
		}

		modifier = {
			factor = 0.8
			devastation = 70
		}

		modifier = {
			factor = 0.8
			devastation = 80
		}

		modifier = {
			factor = 0.01
			has_province_modifier = bioweapon
		}
		
		modifier = {
			factor = 0.35
			OR = {
				has_owner_accepted_culture = no
				controller = {
					is_rival = ROOT
				}
			}
		}

		modifier = {
			factor = 0.75
			NOT = {
				OR = {
					has_terrain = farmlands
					has_terrain = forest
					has_terrain = woods
					has_terrain = grasslands
					has_terrain = drylands
				}
			}
		}
		modifier = {
			factor = 0.35
			OR = {
				has_province_modifier = arid
				has_province_modifier = tropical
				has_province_modifier = arctic
			}
		}
		
		modifier = {
			factor = 0.9
			NOT = {
				controller = {
					mil_tech = 15
				}
			}
		}
		
		modifier = {
			factor = 0.9
			NOT = {
				controller = {
					mil_tech = 30
				}
			}
		}
		
	}
	
	option = {
		name = "province_damage.1.a"
		add_base_tax = -1
	}
}
