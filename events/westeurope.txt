	title = "westeurope.n.1"
	desc = "westeurope.d.1"
	picture = MODERN_UNREST_eventPicture

	major = yes

		is_year = 1840
		NOT = {
			culture = irish
		}

			value = 75

		}

		months = 6
	}
	
	option = {
		name = "westeurope.1.a"
		every_owned_province = {
			limit = {
				continent = europe
				culture = irish
			}
			add_core = IRE
		}
	}
}
