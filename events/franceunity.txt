#province revolt

province_event = {
	id = franceunity.3
	title = "franceunity.n.3"
	desc = "franceunity.d.3"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		is_year = 1815
		NOT = {
			current_age = age_of_great_wars
		}
		NOT = {
			devastation = 1
		}
		
		has_owner_accepted_culture = no
		
		culture_group = french
		is_core = FRA
		owner = {
			NOT = {
				culture_group = french
			}
			NOT = {
				prestige = 65
			}
		}
		exists = FRA
		unrest = 0
	}	

	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = "franceunity.3.a"
		spawn_rebels = {
			type = nationalist_rebels
			size = 3
			friend = FRA
		}
		add_devastation = 15
	}
	option = {
		name = "franceunity.3.b"
		cede_province = FRA
		add_core = FRA
	}
}

#nations join france (NOT BELGIUM OR BURGUNDY)

country_event = {
	id = franceunity.6
	title = "franceunity.n.6"
	desc = "franceunity.d.6"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		is_year = 1815
		ai = yes 
		NOT = {
			current_age = age_of_great_wars
		}
		exists = FRA
		culture_group = french
		is_subject = no
		capital_scope = {
			continent = europe	
		}
		NOT = {
			OR = {
				total_development = 650
				tag = BEL
				tag = FRA
				#tag = BUR
			}
		}
	}	

	mean_time_to_happen = {
		months = 3
	}
	
	option = {
		name = "franceunity.6.a"
		FRA = {
			inherit = ROOT
			ROOT = {
				every_owned_province = {
					add_core = FRA
				}
			}
		}
	}
}
